
pipeline{
    agent any
    stages{
        stage('Build'){
            steps{
                sh 'echo "Hello World"'
                sh '''
                    echo "Multiline shell steps works too"
                    ls -lah
                '''
            }
        }
    }
        stage('Uploading to AWS') {
            steps{
                withAWS(region: 'us-west-2', credentials: 'aws-static') {
                sh 'echo "Uploading content with AWS creds"'
                s3Upload(pathStyleAccessEnabled: true, payloadSigningEnabled: true, file: 'index.html', bucket: 'jenkin-s3-bucket')
            }
        }
    }
}
